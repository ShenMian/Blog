# Linked List

链表.

## 实现
单项链表
```cpp
template <typename T>
class List
{
public:
  struct Node
  {
    T     data;
    Node* next;
  };

  virtual ~List()
  {
    clear();
  }
  
  void insert_after(size_t pos, const T& value)
  {
    if(pos >= size)
      throw std::out_of_range("pos out of range");
    
    // 创建新节点
    auto node = new Node;
    node->data = value;
    node->next = nullptr;

    // 获得 pos 的前一个节点
    auto prev = head;
    for (size_t i = 0; i < pos - 1; i++)
      prev = prev->next;

    // 在 pos 的前一个节点后(即 pos)插入新节点
    node->next = prev->next;
    prev->next = node;

    size++;
  }

  bool empty() const
  {
    return size == 0;
  }

  void clear()
  {
    auto node = head;
    while(node)
    {
      const auto tmp = node;
      node = node->next;
      delete tmp;
    }
    head = nullptr;
    size = 0;
  }

private:
  Node*  head = nullptr;
  size_t size = 0;
}
```
